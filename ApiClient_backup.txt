  /**
   * 鍙戦€?HTTP 璇锋眰鐨勯€氱敤鏂规硶
   */
  private async request<T>(endpoint: string, options: http.HttpRequestOptions, retryCount: number = 0): Promise<ApiResponse<T>> {
    // 绂佺敤Mock鏁版嵁锛屽缁堜娇鐢ㄧ湡瀹炲悗绔疉PI
    if (false) {
      // Mock鏁版嵁鍔熻兘宸茬鐢?
      return this.getMockResponse<T>(endpoint)
    }

    const httpRequest = http.createHttp()
    const url = `${this.baseUrl}${endpoint}`

    try {
      // 璁剧疆璇锋眰澶?
      const headers: Record<string, string> = {
        'Content-Type': 'application/json'
      }

      // 濡傛灉瀛樺湪 JWT Token,娣诲姞鍒?Authorization 澶?
      console.log('=== Token Debug Info ===')
      console.log('jwtToken exists:', !!this.jwtToken)
      console.log('jwtToken value:', this.jwtToken ? this.jwtToken.substring(0, 20) + '...' : 'null')
      console.log('jwtToken length:', this.jwtToken?.length || 0)

      if (this.jwtToken) {
        headers['Authorization'] = `Bearer ${this.jwtToken}`
        console.log('Authorization header set:', headers['Authorization'].substring(0, 30) + '...')
      } else {
        console.warn('鈿狅笍 WARNING: No token available! Request will fail authentication!')
      }
      console.log('========================')

      const requestOptions: http.HttpRequestOptions = {
        method: options.method,
        header: headers,
        extraData: options.extraData,
        readTimeout: this.timeout,
        connectTimeout: this.timeout
      }

      console.log(`Sending ${options.method} request to: ${url}`)

      const response = await httpRequest.request(url, requestOptions)

      if (response.responseCode === 200) {
        try {
          const result = JSON.parse(response.result as string) as ApiResponse<T>
          console.log(`API Response for ${endpoint}:`, result)
          return result
        } catch (parseError) {
          console.error('Failed to parse response:', parseError)
          throw new Error('Invalid response format')
        }
      } else if (response.responseCode === 401) {
        // Token 杩囨湡鎴栨棤鏁?
        this.clearToken()
        throw new Error('Authentication failed')
      } else if (response.responseCode >= 500 && retryCount < this.maxRetries) {
        // 鏈嶅姟鍣ㄩ敊璇紝閲嶈瘯
        console.log(`Retrying request (${retryCount + 1}/${this.maxRetries})...`)
        await this.delay(this.retryDelay)
        return this.request<T>(endpoint, options, retryCount + 1)
      } else {
        throw new Error(`HTTP ${response.responseCode}: ${response.result}`)
      }
    } catch (error) {
      console.error(`Request failed for ${url}:`, error)

      // 缃戠粶閿欒锛岄噸璇?
      if (retryCount < this.maxRetries) {
        console.log(`Retrying request due to network error (${retryCount + 1}/${this.maxRetries})...`)
        await this.delay(this.retryDelay)
        return this.request<T>(endpoint, options, retryCount + 1)
      }

      if (error instanceof Error) {
        throw error
      } else {
        throw new Error(String(error))
      }
    } finally {
      httpRequest.destroy()
    }
  }

  /**
   * 寤惰繜鍑芥暟
   */
  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms))
  }

  /**
   * 鍒涘缓閿欒鍝嶅簲锛堥伩鍏嶄娇鐢╱nknown绫诲瀷锛?
   */
  private createErrorResponse<T>(code: number, message: string): ApiResponse<T> {
    const errorResponse: ApiResponse<T> = {
      code: code,
      message: message,
      data: undefined as T
    }
    return errorResponse
  }

  /**

   */
  async loginByPassword(account: string, password: string): Promise<ApiResponse<LoginResponse>> {
    // 鍒ゆ柇鏄敤鎴峰悕杩樻槸閭锛氬寘鍚獲绗﹀彿鐨勬槸閭锛屽惁鍒欐槸鐢ㄦ埛鍚?
    const isEmail = account.includes('@')
    const requestData: LoginRequest = {
      password: password
    }

    // 鏍规嵁璐﹀彿绫诲瀷璁剧疆瀵瑰簲鐨勫瓧娈?
    if (isEmail) {
      requestData.email = account
    } else {
      requestData.username = account
    }

    const options: http.HttpRequestOptions = {
      method: http.RequestMethod.POST,
      extraData: JSON.stringify(requestData)
    }

    const response = await this.request<string>('/user/login', options)

    // 鍚庣杩斿洖鐨刣ata鏄洿鎺oken瀛楃涓?涓嶆槸瀵硅薄
    console.log('=== Login Response Debug ===')
    console.log('response.code:', response.code)
    console.log('response.message:', response.message)
    console.log('typeof response.data:', typeof response.data)
    console.log('response.data value:', response.data ? (typeof response.data === 'string' ? response.data.substring(0, 30) + '...' : JSON.stringify(response.data)) : 'null')

    // 濡傛灉鐧诲綍鎴愬姛,淇濆瓨 Token (鍚庣鐩存帴杩斿洖token瀛楃涓?
    if (response.code === 1 && response.data && typeof response.data === 'string') {
      console.log('鉁?Saving token to ApiClient...')
      this.setToken(response.data)
      console.log('鉁?Token saved! Current token length:', this.jwtToken.length)
    } else {
      console.error('鉂?Login failed or no token in response!')
      console.error('response.data type:', typeof response.data)
      console.error('response.data value:', response.data)
    }
    console.log('===========================')

    // 涓轰簡鍏煎鍓嶇鍏朵粬浠ｇ爜,灏嗗瓧绗︿覆token鍖呰鎴怢oginResponse瀵硅薄

      code: response.code,
      message: response.message,
      data: new LoginResponse(response.data || '', {} as UserInfo, 7200)
    }

    return loginResponse
  }

  /**
   * 鑾峰彇鐢ㄦ埛淇℃伅
   */
  async getUserInfo(): Promise<ApiResponse<UserInfo>> {
    const options: http.HttpRequestOptions = {
      method: http.RequestMethod.GET
    }

    // 鍚庣杩斿洖鐨勫瓧娈靛悕鏄?totalStep, remainingStep (鍗曟暟)
    // 鍓嶇鏈熸湜鐨勫瓧娈靛悕鏄?totalSteps, remainingSteps (澶嶆暟)
    // 闇€瑕佸湪鎺ユ敹鍒版暟鎹悗杩涜杞崲
    interface BackendUserVo {
      id: number;
      username: string;
      email: string;
      nickname: string;
      sex?: number;
      avatar?: string;
      totalStep: number;        // 鍚庣瀛楁鍚?
      totalPoints: number;
      remainingStep: number;    // 鍚庣瀛楁鍚?
      maxDailySteps?: number;
      updateTime?: string;
    }

    const response = await this.request<BackendUserVo>('/user/info', options)

    // 杞崲瀛楁鍚?
    if (response.code === 1 && response.data) {
      const userInfo: UserInfo = {
        id: response.data.id,
        username: response.data.username,
        email: response.data.email,
        nickname: response.data.nickname,
        sex: response.data.sex,
        avatar: response.data.avatar,
        totalSteps: response.data.totalStep,       // 杞崲: totalStep -> totalSteps
        totalPoints: response.data.totalPoints,
        remainingSteps: response.data.remainingStep // 杞崲: remainingStep -> remainingSteps
      }

      return {
        code: response.code,
        message: response.message,
        data: userInfo
      }
    }

    // 濡傛灉鍝嶅簲澶辫触锛岃繑鍥為敊璇搷搴?
    interface ErrorResponse {
      code: number;
      message: string;
      data: undefined;
    }

    const errorResponse: ErrorResponse = {
      code: response.code,
      message: response.message,
      data: undefined
    }

    return this.createErrorResponse<UserInfo>(response.code, response.message)
  }
